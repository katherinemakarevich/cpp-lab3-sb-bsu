 #include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int main()
{
const double PRICE_APPLE_PER_KG = 1.5;  
double rate;
cout << "Enter the rate (in RUB): ";
cin >> rate;

for (int i = 0; i < 10; i++)
{
string line = "$" + to_string(10 * (i + 1)) + " ";
cout << setw(5) << line;
line = to_string(rate * 10 * (i + 1)) + "RUB ";
cout << setw(15) << line;
line = to_string(floor((rate * 10 * (i + 1)) / PRICE_APPLE_PER_KG)) + " apples";
cout << setw(15) << line << endl;
}
}
